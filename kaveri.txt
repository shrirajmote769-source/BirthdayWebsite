    <script>
        // --- 1. Three.js Background Setup ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
        
        renderer.setSize(window.innerWidth, window.innerHeight);
        document.getElementById('canvas-container').appendChild(renderer.domElement);

        // Lights
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
        scene.add(ambientLight);
        const pointLight = new THREE.PointLight(0xffafcc, 1, 100);
        pointLight.position.set(10, 10, 10);
        scene.add(pointLight);

        // Heart Shape Helper
        function createHeartShape() {
            const x = 0, y = 0;
            const heartShape = new THREE.Shape();
            heartShape.moveTo(x + 5, y + 5);
            heartShape.bezierCurveTo(x + 5, y + 5, x + 4, y, x, y);
            heartShape.bezierCurveTo(x - 6, y, x - 6, y + 7, x - 6, y + 7);
            heartShape.bezierCurveTo(x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19);
            heartShape.bezierCurveTo(x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7);
            heartShape.bezierCurveTo(x + 16, y + 7, x + 16, y, x + 10, y);
            heartShape.bezierCurveTo(x + 7, y, x + 5, y + 5, x + 5, y + 5);
            return heartShape;
        }

        const geometry = new THREE.ExtrudeGeometry(createHeartShape(), {
            depth: 2,
            bevelEnabled: true,
            bevelSegments: 2,
            steps: 2,
            bevelSize: 1,
            bevelThickness: 1
        });

        // Create 25 Hearts
        const hearts = [];
        const material = new THREE.MeshPhysicalMaterial({ 
            color: 0xff6b6b, 
            metalness: 0.1, 
            roughness: 0.2, 
            transmission: 0,
            clearcoat: 1.0
        });

        for(let i=0; i<25; i++) {
            const mesh = new THREE.Mesh(geometry, material);
            // Random Position
            mesh.position.x = (Math.random() - 0.5) * 100;
            mesh.position.y = (Math.random() - 0.5) * 60;
            mesh.position.z = (Math.random() - 0.5) * 50 - 20; // Push back slightly
            
            // Random Rotation
            mesh.rotation.x = Math.PI; // Flip heart up
            mesh.rotation.z = Math.random() * 0.5 - 0.25;

            // Random Scale
            const s = Math.random() * 0.05 + 0.02;
            mesh.scale.set(s, s, s);

            // Custom Animation Data
            mesh.userData = {
                speed: Math.random() * 0.02 + 0.005,
                yOffset: Math.random() * 100,
                xOffset: Math.random() * 100
            };

            scene.add(mesh);
            hearts.push(mesh);
        }

        camera.position.z = 30;

        // Animation Loop
        let time = 0;
        function animate() {
            requestAnimationFrame(animate);
            time += 0.01;

            hearts.forEach(h => {
                // Floating sine wave motion
                h.position.y += Math.sin(time + h.userData.yOffset) * 0.02;
                h.position.x += Math.cos(time + h.userData.xOffset) * 0.01;
                h.rotation.y += 0.005;
            });

            renderer.render(scene, camera);
        }
        animate();

        // Resize Handler
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });


        // --- 2. Step Navigation Logic with GSAP ---
        let currentStep = 1;
        const totalSteps = 5;

        // Initial Fade In
        gsap.to("#step-1", { opacity: 1, y: 0, duration: 1, ease: "power2.out" });
        gsap.from("#step-1 > *", { opacity: 0, y: 20, duration: 0.8, stagger: 0.1, delay: 0.2 });

        function nextStep(stepNumber) {
            const prevId = `#step-${currentStep}`;
            const nextId = `#step-${stepNumber}`;
            const progressBar = document.getElementById("progress-bar");

            // Update Progress Bar
            const progress = ((stepNumber - 1) / (totalSteps - 1)) * 100;
            progressBar.style.width = `${progress}%`;

            // Animate Out Current Step
            gsap.to(prevId, {
                opacity: 0,
                scale: 0.95,
                y: -20,
                duration: 0.5,
                ease: "power2.in",
                onComplete: () => {
                    document.querySelector(prevId).classList.remove('active');
                    const nextEl = document.querySelector(nextId);
                    nextEl.classList.add('active');
                    
                    // Reset starting position for animation
                    gsap.set(nextId, { opacity: 0, scale: 1.05, y: 20 });
                    
                    // Animate In Next Step
                    gsap.to(nextId, {
                        opacity: 1,
                        scale: 1,
                        y: 0,
                        duration: 0.6,
                        ease: "power2.out"
                    });
                    
                    // Stagger children animation
                    gsap.from(`${nextId} > *`, {
                        opacity: 0,
                        y: 15,
                        duration: 0.5,
                        stagger: 0.1,
                        delay: 0.1
                    });
                }
            });

            currentStep = stepNumber;
        }


        // --- 3. Polaroid Tilt Effect (Step 4) ---
        const polaroid = document.getElementById('polaroid');
        const polaroidContainer = document.querySelector('.polaroid-container');

        polaroidContainer.addEventListener('mousemove', (e) => {
            const rect = polaroid.getBoundingClientRect();
            const x = e.clientX - rect.left; // x position within the element.
            const y = e.clientY - rect.top;  // y position within the element.
            
            const centerX = rect.width / 2;
            const centerY = rect.height / 2;
            
            const rotateX = ((y - centerY) / centerY) * -10; // Max rotation deg
            const rotateY = ((x - centerX) / centerX) * 10;

            polaroid.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) scale3d(1.05, 1.05, 1.05)`;
        });

        polaroidContainer.addEventListener('mouseleave', () => {
            polaroid.style.transform = `perspective(1000px) rotateX(0deg) rotateY(0deg) rotate(2deg)`;
        });


        // --- 4. Grand Finale Sequence ---
        function celebrate() {
            const btn = document.getElementById('celebrate-btn');
            const finalMsg = document.getElementById('final-message');

            // 1. Button Fades Out
            gsap.to(btn, { 
                scale: 0.8, 
                opacity: 0, 
                duration: 0.5, 
                onComplete: () => btn.style.display = 'none' 
            });

            // 2. Text Reveal
            finalMsg.classList.remove('hidden');
            gsap.from(finalMsg, { opacity: 0, y: 20, scale: 0.8, duration: 1, ease: "back.out(1.7)" });

            // 3. Three.js Hearts Float Away
            hearts.forEach(h => {
                gsap.to(h.position, {
                    y: h.position.y + 100,
                    x: h.position.x + (Math.random() - 0.5) * 50,
                    rotation: { z: Math.random() * 10, y: Math.random() * 10 },
                    duration: 4 + Math.random() * 2,
                    ease: "power1.in"
                });
            });

            // 4. Confetti Explosion
            const duration = 5000;
            const end = Date.now() + duration;

            // Big Burst First
            confetti({
                particleCount: 150,
                spread: 70,
                origin: { y: 0.6 },
                colors: ['#ec4899', '#ef4444', '#ffffff', '#fbcfe8']
            });

            // Continuous Confetti & Emojis
            (function frame() {
                // Launch two emitters from bottom corners
                confetti({
                    particleCount: 3,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0 },
                    colors: ['#ec4899', '#ffffff']
                });
                confetti({
                    particleCount: 3,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1 },
                    colors: ['#ef4444', '#fbcfe8']
                });

                // Heart Emojis
                if (Date.now() % 15 === 0) {
                     confetti({
                        particleCount: 1,
                        startVelocity: 30,
                        spread: 360,
                        ticks: 60,
                        origin: { x: Math.random(), y: Math.random() * 0.6 },
                        shapes: ['circle'], // canvas-confetti shape limit, using circles mostly but we can try custom shapes if library updated, sticking to colors for safety.
                        scalar: 2, // larger confetti
                        colors: ['#FF0000'] 
                    });
                }

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());
        }
    </script>
